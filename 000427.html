每日创建之自动测试

　　利用ant以及junit进行自动编译和测试。本例只是自动创建了你的测试程序（而非开发的应用程序），然后自动运行测试程序，并将结果发送到你的email信箱中。

第一步：设置每天3点0分进行创建和测试
#crontab -e
0 3 * * * (cd /yourhome/yourproj; ant >/dev/null 2>&1)

第二步：配置ant的build文件

<project name="junit+ant example" default="mail">
  <!-- =================================================================== -->
  <!-- 目标setProps：接收属性定义，进行属性设置                            -->
  <!-- 说明：它与init目标的差别就是它接收来自命令行的属性定义              -->
  <!-- 　　　这么做的目的是为了做大项目时嵌套的ant调用能保持基本环境一致   -->
  <!-- =================================================================== -->
        <target name="setProps" unless="setProps">
                <property name="outdir"   value="./tmp/junitSample" />
                <property name="setProps" vlaue="true" />
        </target>
  <!-- =================================================================== -->
  <!-- 目标init：属性定义　　　　　　　　　　　                            -->
  <!-- 说明：可以理解为全局变量定义。                                      -->
  <!--       其中${lib}目录中放的是编译或运行需要的第三方的jar包，         -->
  <!--       或本项目其它模块生成的jar文件。　　　　　　　　　　　         -->
  <!-- =================================================================== -->
        <target name="init" depends="setProps">
                <property name="lib" value="../lib" />
                <property name="jsdk_lib" value="/path/to/your/jre/lib" />
                <property name="report"   value="${outdir}/report" />
                <property name="report_html"   value="${outdir}/report/html" />
        </target>
  <!-- =================================================================== -->
  <!-- 目标prepare：准备工作 　　　　　　　　　                            -->
  <!-- 说明：这儿就是清除旧的编译结果，生成所需要的目录                    -->
  <!-- =================================================================== -->
        <target name="prepare" depends="clean">
                <mkdir dir="${outdir}" />
                <mkdir dir="${report_html}" />
        </target>
  <!-- =================================================================== -->
  <!-- 目标clean：清除旧的结果 　　　　　　　　                            -->
  <!-- =================================================================== -->
        <target name="clean" depends="init">
                <delete dir="${outdir}" />
        </target>
  <!-- =================================================================== -->
  <!-- 目标compile：进行编译   　　　　　　　　                            -->
  <!-- =================================================================== -->
        <target name="compile" depends="prepare">
                <javac srcdir="./src" destdir="${outdir}" >
                        <classpath>
                                <fileset dir="${lib}">
                                        <include name="**/*.jar" />
                                </fileset>
                                <fileset dir="${jsdk_lib}">
                                        <include name="**/*.jar" />
                                </fileset>

                        </classpath>
                </javac>
        </target>
  <!-- =================================================================== -->
  <!-- 目标test：进行自动测试 　　　　　　　                               -->
  <!-- 说明：将测试结果生成xml文件保存　                                   -->
  <!-- 　　　根据测试结果生成html格式的测试报告                            -->
  <!-- =================================================================== -->
        <target name="test" depends="compile">
                <junit printsummary="true">
                        <formatter type="xml" />
                        <test name="com.hedong.junit.example.AllTests" />
                        <classpath>
                                <pathelement location="${outdir}" />
                                <fileset dir="${lib}">
                                        <include name="**/*.jar" />
                                </fileset>
                                <fileset dir="${jsdk_lib}">
                                        <include name="**/*.jar" />
                                </fileset>
                        </classpath>
                </junit>
                <junitreport todir="${report}">
                        <fileset dir=".">
                                <include name="TEST-*.xml" />
                        </fileset>
                        <report format="noframes" todir="${report_html}" />
                </junitreport>
        </target>
  <!-- =================================================================== -->
  <!-- 目标mail：将测试结果发到信箱中。　　　　                            -->
  <!-- =================================================================== -->
        <target name="mail" depends="test">
                <mail mailhost="smtp.your.mailserver.com" subject="Test build">
                  <from address="yourname@your.mailserver.com"/>
                  <to address="yourname@your.mailserver.com"/>
                  <message>The nightly build/test has completed</message>
                  <fileset dir="${report_html}">
                                <include name="*.html"/>
                  </fileset>
                </mail>
        </target>
</project>

　　本配置文件侧重在自动测试和自动报告方面，其它方面请参考车东的文章。具体的任务的内容，请参考"蚂蚁怒了"，或者ant的主页.
